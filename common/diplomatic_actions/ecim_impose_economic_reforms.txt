force_economic_reforms = {
    groups= {
        power_bloc_leader
    }
    show_in_lens = no

    texture = "gfx/interface/icons/diplomatic_action_icons/force_become_economic_reforms.dds"

    selectable = {
        is_power_bloc_leader = yes
        power_bloc ?= {
            modifier:power_bloc_customs_union_bool = yes
        }
    }

    potential = {
        power_bloc ?= {
            scope:target_country.power_bloc ?= this
        }
    }

    possible = {
        custom_tooltip = {
            text = "FORCE_ECONOMIC_REFORM_ECO_SYSTYEM_PROGRESSIVENESS_DIFFERENCE"
            OR = {
                50 <= {
                    add = {
                        add = active_law:lawgroup_economic_system.type.progressiveness
                        subtract = scope:target_country.active_law:lawgroup_economic_system.type.progressiveness
                        abs = yes
                    }
                }
                50 <= {
                    add = {
                        add = active_law:lawgroup_trade_policy.type.progressiveness
                        subtract = scope:target_country.active_law:lawgroup_trade_policy.type.progressiveness
                        abs = yes
                    }
                }

                50 <= {
                    add = {
                        add = active_law:lawgroup_land_reform.type.progressiveness
                        subtract = scope:target_country.active_law:lawgroup_land_reform.type.progressiveness
                        abs = yes
                    }
                }
            }
        }
        power_bloc = {
            current_cohesion_percentage >= 0.1
        }
        scope:target_country = {
            tenure_in_current_power_bloc_years >= 5
        }
        gdp >= {
            value = scope:target_country.gdp
            multiply = 2
        }
        NOT = {
            scope:target_country ?= {
                is_diplomatic_play_ally_of = ROOT
            }
        }
    }

    accept_effect = {
        scope:target_country = {
            activate_law = root.active_law:lawgroup_economic_system.type
            activate_law = root.active_law:lawgroup_trade_policy.type
            activate_law = root.active_law:lawgroup_land_reform.type
        }
            scope:target_country = {
                change_relations = {
                    country = root
                    value = -10
                }
                if = {
                    limit = {
                        is_subject_of = root
                    }
                    add_liberty_desire = 10
                }
            }


        power_bloc = {
            add_cohesion_percent = scaled_cohesion_cost
        }
    }

    ai = {
        evaluation_chance = {
            value = 0.025 # AI shouldn't use this very often
        }

        will_propose = {
            power_bloc ?= { current_cohesion_percentage >= 0.6 }
            OR = {
                has_attitude = {
                    who = scope:target_country
                    attitude = domineering
                }
                power_bloc ?= { current_cohesion_percentage >= 0.8 }
            }
            OR = {
                AND = {
                    has_attitude = {
                        who = scope:target_country
                        attitude = domineering
                    }
                    power_bloc ?= { current_cohesion_percentage >= 0.8 }
                }
                scope:target_country = {
                    is_country_type = unrecognized
                }
            }
        }

        propose_score = {
            value = 10
        }
    }
}
