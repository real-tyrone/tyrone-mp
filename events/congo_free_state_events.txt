namespace = congo_free_state_events

congo_free_state_events.1 = {
	type = country_event
	placement = scope:afs_capital_scope
	title = congo_free_state_events.1.t
	desc = congo_free_state_events.1.d
	flavor =  congo_free_state_events.1.f

	duration = 3

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		#triggered from the JE
	}

	immediate = {
		set_variable = free_state_created
		set_global_variable = free_state_created
		ruler = {
			save_scope_as = free_state_monarch
		}
		random_scope_state = {
			limit = {
				is_incorporated = no
				state_region = s:STATE_BAS_CONGO
				any_neighbouring_state = {
					is_incorporated = no
					is_in_geographic_region = geographic_region_africa
				}
			}
			save_scope_as = colonial_state
		}
		create_country = {
			origin = ROOT
			tag = AFS
			state = scope:colonial_state
		}
		c:AFS ?= {
			every_neighbouring_state = {
		 		limit = {
		 			owner = ROOT
		 		}
		 		set_state_owner = c:AFS
			}
			save_scope_as = free_state
			activate_law = law_type:law_monarchy
			activate_law = law_type:law_autocracy
			activate_law = law_type:law_subjecthood
			active_law:lawgroup_citizenship ?= { # Subjecthood
				add_amendment = {
					type = amendment_racialized_subjecthood
					sponsor = PREV.ig:ig_armed_forces
				}
			}
			activate_law = law_type:law_extraction_economy
			activate_law = law_type:law_colonial_exploitation
			activate_law = law_type:law_no_schools
			activate_law = law_type:law_no_health_system
			activate_law = law_type:law_outlawed_dissent
			activate_law = law_type:law_no_workers_rights
			activate_law = law_type:law_child_labor_allowed
			activate_law = law_type:law_no_social_security
			activate_law = law_type:law_migration_controls
			activate_law = law_type:law_autocracy
			activate_law = law_type:law_debt_slavery
			add_modifier = {
				name = colonial_administration
				days = -1
			}
			add_technology_researched = malaria_prevention
			every_primary_culture = {
				prev = {
					add_radicals = {
						value = medium_radicals
						culture = prev
					}
					remove_primary_culture = prev
				}
			}
			capital = {
				save_scope_as = afs_capital_scope
			}
		}
		every_primary_culture = { # ROOT's primary cultures
			save_temporary_scope_as = temp_primary_culture
			scope:free_state = {
				set_state_religion = root.religion
				add_primary_culture = scope:temp_primary_culture
			}
		}
		create_diplomatic_pact = {
			country = c:AFS
			type = personal_union
		}
		create_diplomatic_pact = {
			country = c:AFS
			type = raise_payments
		}
	}

	option = { #Let it happen
		name = congo_free_state_events.1.a
		default_option = yes
		set_variable = free_state_made
 		scope:free_state = {
			add_modifier = {
				name = free_state_mort
				days = normal_modifier_time
				multiplier = 0.2
			}
			add_modifier = {
				name = resource_extraction_colony_modifier
				days = very_long_modifier_time
				multiplier = 2
			}
		}
	}
}

		name = congo_free_state_events.5.a

			limit = {
				has_law = law_type:law_monarchy

					has_variable = free_state_monarch

				remove_variable = free_state_created

					name = free_state_monarch_modifier #popularity loss

			}
			add_modifier = {
				name = free_state_monarchy_modifier #legitimacy head of state
				days = normal_modifier_time
			}
		}
	}

	option = { #Free Free State
		name = congo_free_state_events.5.b
		c:AFS ?= {
			make_independent = yes
		}
		if = {
			limit = {
				has_law = law_type:law_monarchy
				ruler = {
					has_variable = free_state_monarch
				}
			}
			ruler = {
				remove_variable = free_state_created
